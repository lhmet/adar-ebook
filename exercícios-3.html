res <- dados_zorra %>%
  gather(., key = variavel, value = valor, -date) %>%
  separate(
    .,
    col = variavel,
    into = c("varname", "hora")
    #sep = "\\."
    ) %>% #.$hora %>% unique()
  mutate(.,
         hora = ifelse(
           nchar(hora) >= 3,
           as.numeric(hora)/100,
           as.numeric(hora)),
         hora = paste0(hora,":", "00:00"),
         date = as.Date(date, "%d-%m-%Y")
         ) %>%
  unite(., col = date, date, hora, sep = " ") %>%
  mutate(., date = as.POSIXct(date)) %>%
  spread(., varname, valor)
